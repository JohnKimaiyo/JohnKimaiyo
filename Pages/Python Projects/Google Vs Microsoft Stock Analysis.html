import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt

# Define the stock symbols and time period
symbols = ['GOOGL', 'MSFT']
start_date = '2020-01-01'
end_date = '2023-01-01'

# Fetch historical stock data using yfinance
stock_data = yf.download(symbols, start=start_date, end=end_date)['Adj Close']

# Plotting
plt.figure(figsize=(12, 6))
for symbol in symbols:
    plt.plot(stock_data.index, stock_data[symbol], label=symbol)

plt.title('Stock Price Comparison: Google vs Microsoft')
plt.xlabel('Date')
plt.ylabel('Stock Price')
plt.legend()
plt.show()
