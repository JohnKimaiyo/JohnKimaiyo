import yfinance as yf
import pandas as pd
import matplotlib.pyplot as plt

# Define the stock symbols for NVIDIA and Cisco
nvidia_symbol = "NVDA"
cisco_symbol = "CSCO"

# Define the time period for stock price data (e.g., last 5 years)
start_date = "2019-01-01"
end_date = "2023-01-01"

# Fetch stock price data using yfinance
nvidia_data = yf.download(nvidia_symbol, start=start_date, end=end_date)
cisco_data = yf.download(cisco_symbol, start=start_date, end=end_date)

# Extract the 'Close' prices
nvidia_close_prices = nvidia_data['Close']
cisco_close_prices = cisco_data['Close']

# Create a DataFrame to store the stock prices
stock_prices_df = pd.DataFrame({
    'NVIDIA': nvidia_close_prices,
    'Cisco': cisco_close_prices
})

# Plot the stock prices
stock_prices_df.plot(title='NVIDIA vs Cisco Stock Prices', ylabel='Stock Price (USD)', xlabel='Date')
plt.show()

# Calculate and display summary statistics
summary_statistics = stock_prices_df.describe()
print(summary_statistics)

# Calculate and display the correlation between NVIDIA and Cisco stock prices
correlation = stock_prices_df['NVIDIA'].corr(stock_prices_df['Cisco'])
print(f"Correlation between NVIDIA and Cisco stock prices: {correlation}")

# Calculate and display the percentage change in stock prices
percentage_change = stock_prices_df.pct_change()
print(percentage_change.head())
